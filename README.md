# nandos-resturants

# setup

1. npm i

# ios setup:

1. cd ios && pod install && cd ..

# run ios

1. npm run ios

# android setup:

1. create local.properties file in android folder which contains a reference to your android sdk e.g. 'sdk.dir = /Users/[User]/Library/Android/sdk'

# run android

1. npm run android
